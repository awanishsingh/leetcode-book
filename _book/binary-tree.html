Binary Tree Preorder Traversal

Depth-first search: get the node value when we see the node in the first time.

def preorder(root):
    values = []

    stack, current = [], root
    while stack or current:
        if current:
            values.append(current.val)
            stack.append(current)
            current = current.left
        else:
            current = stack.pop()
            current = current.right

    return values

Breadth-first search

def preorder(root):
    values = []

    if not root:
        return []

    stack = [root]
    while stack:
        current = stack.pop()
        values.append(current.val)
        if current.right:
            stack.append(current.right)
        if current.left:
            stack.append(current.left)

    return values

Binary Tree Postorder Traversal

Depth-first search: get the node value when we traverse back from the right subtree.

def postorder(root):
    values = []

    previous = None
    stack, current = [], root
    while stack or current:
        if current:
            stack.append(current)
            current = current.left
        else:
            # Need to preserve the node when we traverse the right subtree.
            peek = stack[-1]
            if not peek.right or peek.right == previous:
                values.append(peek.val)
                previous = stack.pop()
            else:
                current = peek.right
    return values

Breadth-first search: use two stacks to reverse the visit order.

def postorder(root):
    values = []

    stack2 = []

    stack1 = [root] if root else []  # Aware edge case.
    while stack1:
        current = stack1.pop()
        stack2.append(current)

        if current.left:
            stack1.append(current.left)
        if current.right:
            stack1.append(current.right)

    while stack2:
        values.append(stack2.pop().val)

    return values
